actor User
participant MainScreen
participant CourtsScreen
participant ResultsScreen
participant NoAvailableCourtScreen
participant DataBaseManager

User->MainScreen:goTo()
activate MainScreen
User->MainScreen:chooseSport
MainScreen->CourtsScreen:goTo()
activate CourtsScreen
deactivate MainScreen
alt Apply Filters
User->CourtsScreen:applyFilter()
CourtsScreen->DataBaseManager:searchCourts()
activate DataBaseManager
DataBaseManager-->CourtsScreen:returnCourts()
deactivate DataBaseManager
CourtsScreen-->*ResultsScreen:<create>
CourtsScreen->ResultsScreen:display()
deactivate CourtsScreen
activate ResultsScreen
User->ResultsScreen:chooseCourt
ResultsScreen->DataBaseManager:submitOption()
activate DataBaseManager
DataBaseManager-->ResultsScreen:returnOption()
deactivate DataBaseManager
deactivate ResultsScreen

else Search By Region
User->CourtsScreen:searchByRegion()
activate CourtsScreen
CourtsScreen->DataBaseManager:searchCourts()
activate DataBaseManager
DataBaseManager-->CourtsScreen:returnCourts()
deactivate DataBaseManager
CourtsScreen->ResultsScreen:display()
activate ResultsScreen
deactivate ResultsScreen
deactivate CourtsScreen

else No Available Court
User->CourtsScreen:searchByRegion()
activate CourtsScreen
CourtsScreen->DataBaseManager:searchCourts()
activate DataBaseManager
DataBaseManager-->CourtsScreen:returnCourts()
deactivate DataBaseManager
CourtsScreen->*NoAvailableCourtScreen:<create>
CourtsScreen->NoAvailableCourtScreen:display()
activate NoAvailableCourtScreen
deactivate NoAvailableCourtScreen
deactivate CourtsScreen
end
