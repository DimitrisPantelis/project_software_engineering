actor User

participant Homescreen
participant RegScreen
participant Account
participant ExistAccScreen
participant LoginScreen
participant WeakPassScreen
participant ConfEmailScreen
participant ConfRegScreen
participant DataBaseManager


User->Homescreen:selectSignUp()
activate Homescreen
Homescreen->RegScreen:goTo()
deactivate Homescreen
activate RegScreen
User->RegScreen:enterDetails
User->RegScreen:clickSubmitButton()

RegScreen->DataBaseManager:checkDetails()
activate DataBaseManager
RegScreen<--DataBaseManager:returnDetails
deactivate DataBaseManager

alt ValidateDetails
RegScreen-->*Account:<create>
activate Account
RegScreen->Account:display()
deactivate Account
deactivate RegScreen
else NonValidateDetails
User->RegScreen:enterExistingEmail
activate RegScreen
RegScreen-->*ExistAccScreen:<create>
RegScreen->ExistAccScreen:display()
activate ExistAccScreen

deactivate RegScreen
ExistAccScreen-->*LoginScreen:<create>
activate LoginScreen

ExistAccScreen->LoginScreen:goTo()
deactivate LoginScreen
User->RegScreen:enterWeakPass
activate RegScreen
RegScreen-->*WeakPassScreen:<create>
deactivate RegScreen
ExistAccScreen->WeakPassScreen:display()
activate WeakPassScreen

ExistAccScreen<--WeakPassScreen:enterStrongerPass
deactivate WeakPassScreen
end

ExistAccScreen-->*ConfEmailScreen:<create>
ExistAccScreen->ConfEmailScreen:display()
activate ConfEmailScreen

deactivate ExistAccScreen
User->ConfEmailScreen:clickLinkButton()
ConfEmailScreen->ConfEmailScreen:checkvalidateLink()
ConfEmailScreen-->*ConfRegScreen:<create>
ConfEmailScreen->ConfRegScreen:goTo()
deactivate ConfEmailScreen
activate ConfRegScreen
ConfRegScreen->ConfRegScreen:successReg()
